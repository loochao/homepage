#include "object.h" 
#if defined(__APPLE__)
    #include <Glut/glut.h>
#else
    #include <GL/glut.h>
#endif
#include <iostream>
#include <stdlib.h>
#include <unistd.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <pthread.h>
#include <signal.h>
#include <errno.h>
#include <string.h>
#include <stdio.h>
#include "terminal.h" 
#include "scene.h" 
#include "cube.h"

unsigned int button = 0;
ScenePtr scene = ScenePtr (new Scene);
TerminalPtr terminal = TerminalPtr (new Terminal);
int pipes[3][2];
pthread_t out, err;
pthread_mutex_t terminal_lock;

/* Execute the requested application in the child process */
void child(void)
{
    std::string path = "/bin/bash";
    std::string arg  = "-i";
    if (execl(path.c_str(), path.c_str(), arg.c_str(), NULL) == -1) {
        perror("execl failed");
        exit(EXIT_FAILURE);
    }
}


/* reads from in writes to out
 * returns 0 if at end of file, 1 otherwise */
int copy(int in, int out)
{
    int len;
    char buffer[1024];
    if ((len = read(in, buffer, 1024)) == -1) {
        /* Interupted system call - SIGCHLD will raise this, so ignore */
        if (errno == EINTR)
            return 0;
        perror("read failed");
        exit(EXIT_FAILURE);     
    } else if (len > 0) {
        std::string line(buffer, len);
        if (pthread_mutex_lock(&terminal_lock) == -1) {
            perror("pthread_mutex_lock failed");
            exit(EXIT_FAILURE);
        }
        terminal->print (line);
        if (pthread_mutex_unlock(&terminal_lock) == -1) {
            perror("pthread_mutex_unlock failed");
            exit(EXIT_FAILURE);
        }
        return 1;
    }
    return 0;
}

void handle_stdin()
{
    while(copy(0, pipes[0][1]))
        ;
}
void *handle_stdout(void *arg)
{
    while(copy(pipes[1][0], 1))
        ;
    return NULL;
}
void *handle_stderr(void *arg)
{
    while(copy(pipes[2][0], 2))
        ;
    return NULL;
}

/* separate thread to handle cleanup and exit */
void *handle_terminate(void *arg)
{
    if (pthread_mutex_init(&terminal_lock, NULL) == -1) {
        perror("pthread_mutex_init failed");
        exit(EXIT_FAILURE);
    }   
    /* allow stdin and stdout to terminate when their pipes close */
    if (pthread_join(out, NULL) == -1) {
        perror("pthread_join failed");
        exit(EXIT_FAILURE);
    }
    if (pthread_join(err, NULL) == -1) {
        perror("pthread_join failed");
        exit(EXIT_FAILURE);
    }

    if (pthread_mutex_destroy(&terminal_lock) == -1) {
        perror("pthread_mutex_destroy failed");
        exit(EXIT_FAILURE);
    }
    
    /* hard exit, as we have to kill the main thread */
    exit(EXIT_SUCCESS);
}

/* signal handler for child exit */
void handle_child(int sig)
{
    pthread_t term;
    /* spawn a new thread to handle cleanup and exit */
    if (pthread_create(&term, NULL, &handle_terminate, NULL) == -1) {
        perror("pthread_create failed");
        exit(EXIT_FAILURE);
    }
}

/* setup exit handlers, log file and mutex,
 * launch worker threads and handle stdin */
void parent(void)
{
    /* setup our child signal handler */
    signal(SIGCHLD, handle_child);

    /* spawn threads to handle stdout and stderr */ 
    if (pthread_create(&out, NULL, &handle_stdout, NULL) == -1) {
        perror("pthread_create failed");
        exit(EXIT_FAILURE);
    }
    if (pthread_create(&err, NULL, &handle_stderr, NULL) == -1) {
        perror("pthread_create failed");
        exit(EXIT_FAILURE);
    }
    /* handle stdin ourselves */
    handle_stdin();
}

void on_key_press (unsigned char key, int x, int y) {
    terminal->key_press (key);
    glutPostRedisplay();
}
void on_special_key_press (int key, int x, int y) {
    unsigned char k = 0;
    switch (key) {
    case GLUT_KEY_UP:    k = 'i'-'a' + 1; break; // Control-i
    case GLUT_KEY_DOWN:  k = 'j'-'a' + 1; break; // Control-j
    case GLUT_KEY_LEFT:  k = 'b'-'a' + 1; break; // Control-b
    case GLUT_KEY_RIGHT: k = 'f'-'a' + 1; break; // Control-f
    case GLUT_KEY_HOME:  k = 'a'-'a' + 1; break; // Control-a
    case GLUT_KEY_END:   k = 'e'-'a' + 1; break; // Control-e
    default:             k = 0;           break;
    }
    terminal->key_press (k);
    glutPostRedisplay();
}
void on_mouse_button (int b, int state, int x, int y) {
    int modifier = glutGetModifiers();
    switch (b) {
    case GLUT_LEFT_BUTTON:   button = 1;  break;
    case GLUT_MIDDLE_BUTTON: button = 2;  break;
    case GLUT_RIGHT_BUTTON:  button = 3;  break;
    default:                 button = 0;  break;
    }
    switch (state) {
    case GLUT_UP:
        scene->button_release (button, x, y, modifier);
        glutPostRedisplay();
        break;
    case GLUT_DOWN:
        scene->button_press (button, x, y, modifier);
        glutPostRedisplay();
        break;
    }
}
void on_mouse_move (int x, int y) {
    scene->mouse_motion (button, x, y);
    glutPostRedisplay();
}
void display (void) {
    glClearColor (1,1,1,1);
    glClear (GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glBlendFunc (GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
    if (pthread_mutex_lock(&terminal_lock) == -1) {
        perror("pthread_mutex_lock failed");
        exit(EXIT_FAILURE);
    }
    terminal->render();
    if (pthread_mutex_unlock(&terminal_lock) == -1) {
        perror("pthread_mutex_lock failed");
        exit(EXIT_FAILURE);
    }
    scene->render();
    glutSwapBuffers();
}
void reshape (int width, int height) {
    glViewport (0, 0, width, height);
    glutPostRedisplay();
}
void timeout (int value) {
    glutPostRedisplay();
    glutTimerFunc (int(1000.0/4), timeout, 1);
}
void terminal_event (Terminal *terminal, std::string input) {
    input += "\n";
    int res = write (pipes[0][1], input.c_str(), input.size());
    if (not res) { }
}

int main (int argc, char **argv) {
    /* Create pipes */
    for (int i = 0; i < 3; i++) {
        if (pipe(pipes[i]) == -1) {
            perror("pipe failed");
            exit(EXIT_FAILURE);
        }
    }
    int pid = fork();
    if (pid == -1) {
        perror("fork failed");
        exit(EXIT_FAILURE);
    }   
    if (pid == 0) { /* child */
        close(pipes[0][1]); /* close unused pipe ends */
        close(pipes[1][0]);
        close(pipes[2][0]);

        dup2(pipes[0][0], 0); /* patch up stdin/out/err */
        dup2(pipes[1][1], 1);
        dup2(pipes[2][1], 2);
        child();

    } else { /* parent */
        close(pipes[0][0]); /* close unused pipe ends */
        close(pipes[1][1]);
        close(pipes[2][1]);

        /* setup our child signal handler */
        signal(SIGCHLD, handle_child);
        /* spawn threads to handle stdout and stderr */ 
        if (pthread_create(&out, NULL, &handle_stdout, NULL) == -1) {
            perror("pthread_create failed");
            exit(EXIT_FAILURE);
        }
        if (pthread_create(&err, NULL, &handle_stderr, NULL) == -1) {
            perror("pthread_create failed");
            exit(EXIT_FAILURE);
        }

        glutInit (&argc, argv);
        glutInitDisplayMode (GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);
        glutCreateWindow ("Terminal");
        glutMouseFunc (on_mouse_button);
        glutMotionFunc (on_mouse_move);
        glutReshapeFunc (reshape);
        glutDisplayFunc (display);
        glutKeyboardFunc (on_key_press);
        glutSpecialFunc (on_special_key_press);
        glutReshapeWindow (600,400);
        glutTimerFunc (int(1000.0/4), timeout, 1);

        CubePtr cube = CubePtr (new Cube);
        scene->set_bg_color (1,1,1,0);
        scene->set_br_color (0,0,0,0);
        scene->set_gravity (-1,-1);
        scene->set_position (10,10);
        scene->set_size (.75,.75);
        scene->set_radius (0);
        scene->set_zoom (1);
        scene->add (cube);

        terminal->set_margin    (Margin(5,5,5,5));
        terminal->set_radius    (0);
        terminal->set_size      (1, 1);
        terminal->set_gravity   (0,0);
        terminal->set_position  (0,0);
        terminal->set_fg_color  (0,0,0,1);
        terminal->set_bg_color  (1,1,.9,1);
        terminal->set_br_color  (0,0,0,0);
        terminal->set_prompt ("");
        terminal->connect (terminal_event);

        glutMainLoop();
    }
    
    /* We shouldn't really end up here
     * but we can if we are terminated directly
     * rather than waiting for the child to exit */
    return EXIT_SUCCESS;
}
