#include "object.h" 
#if defined(__APPLE__)
    #include <Glut/glut.h>
#else
    #include <GL/glut.h>
#endif
#include "colorbar.h" 

std::vector<Colorbar *> colorbars;

void display (void) {
    glClearColor (1,1,1,1);
    glClear (GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glBlendFunc (GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
    for (unsigned int i=0; i<colorbars.size(); i++)
        colorbars[i]->render();
    glutSwapBuffers();
}
void reshape (int width, int height) {
    glViewport (0, 0, width, height);
    glutPostRedisplay();
}

Colorbar *make_colorbar(std::string title, Colormap *cmap, Size size, Position position) {
    Colorbar *colorbar;
    colorbar = new Colorbar();
    colorbar->set_title (title);
    colorbar->set_gravity (1,0);
    colorbar->set_position (size);
    colorbar->set_size (position);
    colorbar->set_cmap (cmap);
    return colorbar;
}

int main (int argc, char **argv) {
    glutInit (&argc, argv);
    glutInitDisplayMode (GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);
    glutCreateWindow ("Colorbars");
    glutReshapeFunc (reshape);
    glutDisplayFunc (display);
    glutReshapeWindow (400,400);

    Colormap *cmap = new Colormap();
    cmap->append (0.0, Color(1,1,1,1));
    cmap->append (1.0, Color(1,0,1,1));

    float x  = 0.075;
    float dx = (1-x)/6.0;

    colorbars.push_back (make_colorbar("Ice",        Colormap::Ice(),        Position(x+0*dx,.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("Fire",       Colormap::Fire(),       Position(x+1*dx,.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("Hot",        Colormap::Hot(),        Position(x+2*dx,.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("IceAndFire", Colormap::IceAndFire(), Position(x+3*dx,.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("Grey",       Colormap::Grey(),       Position(x+4*dx,.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("User Defined", cmap,                 Position(x+5*dx,.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("Red",        Colormap::Red(),        Position(x+0*dx,-.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("LightRed",   Colormap::LightRed(),   Position(x+1*dx,-.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("Green",      Colormap::Green(),      Position(x+2*dx,-.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("LightGreen", Colormap::LightGreen(), Position(x+3*dx,-.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("Blue",       Colormap::Blue(),       Position(x+4*dx,-.25), Size (20,.4)));
    colorbars.push_back (make_colorbar("LightBlue",  Colormap::LightBlue(),  Position(x+5*dx,-.25), Size (20,.4)));

    glutMainLoop();
    return 0;
}
