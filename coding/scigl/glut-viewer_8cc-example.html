#include "object.h" 
#include <sys/time.h>
#if defined(__APPLE__)
    #include <Glut/glut.h>
#else
    #include <GL/glut.h>
#endif
#include <cstdlib>
#include "fparser.hh"
#include "scene.h" 
#include "terminal.h" 
#include "colorbar.h" 
#include "surface.h"
#include "frame.h"

ScenePtr scene = ScenePtr (new Scene);
TerminalPtr terminal = TerminalPtr (new Terminal);
ColorbarPtr colorbar = ColorbarPtr (new Colorbar);
ColormapPtr cmap = Colormap::Hot();
FramePtr frame = FramePtr (new Frame);
SurfacePtr surface = SurfacePtr (new Surface);
unsigned int button = 0;
unsigned int fps = 40;
int width = 64;
int height = 64;
GLdouble *values = new GLdouble[width*height];
float min_x = -1;
float max_x = +1;
float min_y = -1;
float max_y = +1;
struct timeval time_start, now;
FunctionParser fparser;
int parser_result;


void update () {
    if (parser_result >= 0)
        return;
    gettimeofday (&now, NULL);
    float t = ((now.tv_sec-time_start.tv_sec)*1000.0 +
               (now.tv_usec-time_start.tv_usec)/1000.0)/1000.0;
    double vals[3];
    float min = 0;
    float max = 0;
    vals[2] = t;
    for (int j=0; j<height; j++) {
        for (int i=0; i<width; i++) {
            vals[0] = min_x + i/GLdouble(width)*(max_x-min_x);
            vals[1] = min_y + j/GLdouble(height)*(max_y-min_y);
            GLdouble value = fparser.Eval(vals);
            values[j*width+i] = value;
            if (value > max)
                max = value;
            if (value < min)
                min = value;
        }
    }
    /*
    if (surface) {
        if (max > min) {
            surface->set_size     (1,1, .5/(max-min));
            surface->set_position (0,0, -min/(max-min)-.5);
        } else {
            surface->set_size     (1,1,.5);
            surface->set_position (0,0,0);
        }
    }
    if (frame) {
        frame->set_z_range (Range (2*max, 2*min, 4, 20));
        frame->set_x_range (Range (min_x, max_x, 4, 20));
        frame->set_y_range (Range (min_y, max_y, 4, 20));
    }
    cmap->scale(min,max);
    */
    surface->update();
    colorbar->update();
}
void parse (std::string function) {
    parser_result = fparser.Parse(function, "x,y,t");
    if (parser_result >= 0) {
        terminal->print (std::string(fparser.ErrorMsg()) + "\n\n");
        return;
    }
    gettimeofday (&time_start, NULL);
    update();
}
void on_key_press (unsigned char key, int x, int y) {
    if (key == 27)
        if (terminal->get_visible())
            terminal->hide (250);
        else
            terminal->show (250);
    else
        terminal->key_press (key);
    glutPostRedisplay();
}

void on_special_key_press (int key, int x, int y) {
    unsigned char k = 0;
    switch (key) {
    case GLUT_KEY_F1:
        if (frame->get_visible())
            frame->hide (250);
        else                 
            frame->show (250);        
        break;
    case GLUT_KEY_F2:
        if (colorbar->get_visible()) colorbar->hide (250);
        else                         colorbar->show (250);        
        break;
    case GLUT_KEY_UP:    k = 'i'-'a' + 1; break; // Control-i
    case GLUT_KEY_DOWN:  k = 'j'-'a' + 1; break; // Control-j
    case GLUT_KEY_LEFT:  k = 'b'-'a' + 1; break; // Control-b
    case GLUT_KEY_RIGHT: k = 'f'-'a' + 1; break; // Control-f
    case GLUT_KEY_HOME:  k = 'a'-'a' + 1; break; // Control-a
    case GLUT_KEY_END:   k = 'e'-'a' + 1; break; // Control-e
    default:             k = 0;           break;
    }
    terminal->key_press (k);
    glutPostRedisplay();
}
void on_mouse_button (int b, int state, int x, int y) {
    int glut_modifier = glutGetModifiers();
    int modifier = 0;
    if (glut_modifier & GLUT_ACTIVE_SHIFT)
        modifier |= 0x1;
    if (glut_modifier & GLUT_ACTIVE_CTRL)
        modifier |= 0x2;
    if (glut_modifier & GLUT_ACTIVE_ALT)
        modifier |= 0x4;

    switch (b) {
    case GLUT_LEFT_BUTTON:   button = 1;  break;
    case GLUT_MIDDLE_BUTTON: button = 2;  break;
    case GLUT_RIGHT_BUTTON:  button = 3;  break;
    default:                 button = 0;  break;
    }
    switch (state) {
    case GLUT_UP:
        scene->button_release (button, x, y, modifier);
        glutPostRedisplay();
        break;
    case GLUT_DOWN:
        scene->button_press (button, x, y, modifier);
        glutPostRedisplay();
        break;
    }
}
void on_mouse_move (int x, int y) {
    scene->mouse_motion (button, x, y);
    glutPostRedisplay();
}
void display (void) {
    glClearColor (1,1,1,1);
    glClear (GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
    glBlendFunc (GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
    terminal->render();
    update();
    scene->render ();
    glutSwapBuffers();
}
void reshape (int width, int height) {
    glViewport (0,0,width, height);
    glutPostRedisplay();
}
void timeout (int value) {
    glutPostRedisplay();
    glutTimerFunc (int(1000.0/fps), timeout, 1);
}
void
terminal_event (Terminal *terminal, std::string input) {
    parse (input);
}

int main (int argc, char **argv) {
    glutInit (&argc, argv);
    glutInitDisplayMode (GLUT_DOUBLE | GLUT_RGB | GLUT_DEPTH);
    glutCreateWindow ("OpenGL viewer");
    glutReshapeFunc (reshape);
    glutDisplayFunc (display);
    glutMouseFunc (on_mouse_button);
    glutMotionFunc (on_mouse_move);
    glutKeyboardFunc (on_key_press);
    glutSpecialFunc (on_special_key_press);
    glutReshapeWindow (600,400);
    glutTimerFunc (int(1000.0/fps), timeout, 1);

    scene->set_bg_color (0,0,0,0);
    terminal->set_prompt ("f(x,y,t) = ");
    terminal->connect (terminal_event);

    cmap->scale(-1,1);

    colorbar->set_cmap (cmap);
    colorbar->set_gravity (-1,0);
    colorbar->set_position (.1,0);
    colorbar->set_size (20,.75);
    colorbar->set_sizeable (false);

    DataPtr data = DataPtr (new Data);
    data->set (values, width, height, 1, GL_DOUBLE, sizeof(GLdouble));
    surface->set_colormap(cmap);
    surface->set_z_c_data (data, data);
    surface->set_size(1,1,.25);
    parse ("(1-x*3/2+(x*3)^5+(y*3)^3)*exp(-x*x*9-y*y*9)*cos(t)");
    scene->add (colorbar);
    scene->add (surface);    
    scene->add (frame);
    scene->set_zoom (1.75);

    terminal->print ("\033[1m3D Function Viewer\033[0m\n");
    terminal->print ("Using warp fparser library ");
    terminal->print ("(\033[4;38;5;21mhttp://warp.povusers.org/FunctionParser\033[0m)\n\n");
    terminal->print ("\033[34;1mKeyboard:\033[0m\n");
    terminal->print ("\033[34m  Escape: toggle terminal\033[0m\n");
    terminal->print ("\033[34m  F1:     toggle frame\033[0m\n");
    terminal->print ("\033[34m  F2:     toggle colorbar\033[0m\n\n");
    terminal->print ("\033[34;1mMouse:\033[0m\n");
    terminal->print ("\033[34m  Button 1:       Rotate figure\033[0m\n");
    terminal->print ("\033[34m  Shift+Button 1: Move figure\033[0m\n");
    terminal->print ("\033[34m  Button 2:       Zoom figure\033[0m\n\n");
    glutMainLoop();
    return 0;
}
